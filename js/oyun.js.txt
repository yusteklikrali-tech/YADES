// Sayı tahmini oyunu
let hedefSayi = Math.floor(Math.random() * 100) + 1;

function ebaOyun() {
  alert("EBA'nın oyun bölümüne yönlendiriliyorsun.");
  window.open("https://eba.gov.tr", "_blank");
}

function sohbetOyun() {
  document.querySelector(".secenekler").style.display = "none";
  document.getElementById("sohbet-ekrani").style.display = "block";
}

function kontrolEt() {
  const tahmin = parseInt(document.getElementById("tahmin").value);
  const sonuc = document.getElementById("sonuc");

  if (isNaN(tahmin)) {
    sonuc.textContent = "Lütfen geçerli bir sayı gir!";
    sonuc.style.color = "orange";
    return;
  }

  if (tahmin === hedefSayi) {
    sonuc.textContent = `🎉 Tebrikler! Doğru bildin: ${hedefSayi}`;
    sonuc.style.color = "green";
  }
  else if (tahmin < hedefSayi) {
    sonuc.textContent = "❌ Daha büyük bir sayı dene!";
    sonuc.style.color = "blue";
  }
  else {
    sonuc.textContent = "❌ Daha küçük bir sayı dene!";
    sonuc.style.color = "red";
  }

  document.getElementById("tahmin").value = "";
}

function yeniOyun() {
  hedefSayi = Math.floor(Math.random() * 100) + 1;
  document.getElementById("sonuc").textContent = "";
  document.getElementById("tahmin").value = "";
}

function geri() {
  window.location.href = "index.html";
}
